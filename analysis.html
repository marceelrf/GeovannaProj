data_fix <- data %>% 
  mutate(across(ends_with("h"),
                \(x) str_remove(x,pattern = "\\<|\\>")
                )
         ) %>% 
  pivot_longer(cols = ends_with("h"),
               names_to = "Time",
               values_to = "Values") %>% 
  mutate(Values = as.numeric(Values),
         Time = factor(Time,
                       levels = c("0h","6h","12h","18h","24h")))
